DATA:
  lista 0xA0
  n 5
  res 0
  aux 0
CODE:
siguiente:
  MOV B,(lista)
  MOV A
  INK B
  MOV B,(B)
  CMP A,B
  JGE comparar
  MOV (aux),A
  MOV A,B
  MOV B,(aux)
comparar:
  SUB A,B
  CMP A,(res)
  JLE 0x0a
  MOV (res),A
revisar:
  MOV A,(n)
  SUB A,-0b00
  MOV (n),A
  INC (lista)
  CMP A,0b01
  JGT siguiente
